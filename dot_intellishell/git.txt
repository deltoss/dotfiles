# --------------------------------------------------------------
#   Commands
# --------------------------------------------------------------

# [alias:gco] Checkout a branch
git checkout {{branch}}

# --------------------------------------------------------------
#   Completions
# --------------------------------------------------------------

# A global completion for any `{{branch}}` variable.
$ branch: git branch --format='%(refname:short)'
$ b: git branch --format='%(refname:short)'

# A global completion for any `{{remote}}` variable.
$ remote: git remote
$ r: git remote

# --------------------------------------------------------------
#   Renames
# --------------------------------------------------------------

# [alias:grr] Rename a remote branch.
git push {{remote}} {{remote}}/{{branch}}:refs/heads/{{new-name}} :{{branch}}
