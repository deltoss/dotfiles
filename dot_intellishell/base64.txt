# [alias:base64] Converts a string to base 64
[Convert]::ToBase64String([System.Text.Encoding]::UTF8.GetBytes('{{{textToConvert}}}'))

# [alias:basicauth] Converts a username and password to base 64 for basic authentication
Write-Host "Authentication: Basic $([Convert]::ToBase64String([System.Text.Encoding]::UTF8.GetBytes('{{username}}:{{{password}}}')))"
